### 1.1 Поэтапный подход

Мы развёртываем Вечный Двигатель в четырех отдельных фазах, каждая из которых основывается на предыдущей. Это минимизирует риск при проверке допущений.

```
ФАЗА 1: ФУНДАМЕНТ (Недели 1-4)
├── Цель: Доказать работоспособность CORE-HODL
├── Уровень риска: МИНИМАЛЬНЫЙ (только спот)
├── Капитал: $5,000-$10,000
└── Результат: Установлены долгосрочные позиции

ФАЗА 2: ВАЛИДАЦИЯ (Недели 5-12)
├── Цель: Тестировать все движки в демо
├── Уровень риска: НИЗКИЙ (бумажная торговля)
├── Капитал: $0 (только демо)
└── Результат: Стратегии проверены, баги исправлены

ФАЗА 3: АКТИВАЦИЯ (Недели 13-20)
├── Цель: Полная система в реальном времени с небольшим капиталом
├── Уровень риска: СРЕДНИЙ (реальные деньги, небольшой размер)
├── Капитал: $20,000-$50,000
└── Результат: Все движки работают

ФАЗА 4: МАСШТАБИРОВАНИЕ (Месяцы 6-12)
├── Цель: Масштабирование до целевого распределения
├── Уровень риска: УПРАВЛЯЕМЫЙ (полная система)
├── Капитал: $100,000+
└── Результат: Автономная работа
```

### 1.2 Критерии успеха по фазам

| Фаза | Критерии входа | Метрики успеха | Критерии выхода |
|------|---------------|----------------|-----------------|
| **Фаза 1** | Нет | Позиции открыты, работающая инфраструктура | Позиции >90% от целевых |

# deployment/phase1_core_hodl.py

class CoreHodlDeployment:
    def __init__(self, capital):
        self.total_capital = capital
        self.btc_target = capital * 0.40  # 40% BTC
        self.eth_target = capital * 0.20  # 20% ETH
        self.stables = capital * 0.40     # 40% USDT (для будущих движков)
        
    def execute_dca(self, days=14):
        """
        Размещение капитала за 14 дней для снижения риска тайминга
        """
        daily_btc = self.btc_target / days
        daily_eth = self.eth_target / days
        
        for day in range(1, days + 1):
            print(f"День {day}/{days}: Размещение...")
            
            # Покупка BTC
            order_btc = self.place_market_buy(
                symbol='BTC/USDT',
                notional=daily_btc
            )
            
            # Покупка ETH
            order_eth = self.place_market_buy(
                symbol='ETH/USDT',
                notional=daily_eth
            )
            
            # Логирование размещения
            self.log_deployment(day, order_btc, order_eth)
            
            # Ожидание 24 часа
            time.sleep(86400)
            
    def verify_allocation(self):
        """
        Проверка достижения целевого распределения
        """
        balances = self.get_balances()
        
        btc_value = balances['BTC'] * self.get_btc_price()
        eth_value = balances['ETH'] * self.get_eth_price()
        total_value = btc_value + eth_value
        
        btc_pct = btc_value / total_value
        eth_pct = eth_value / total_value
        
        print(f"Текущее распределение:")
        print(f"  BTC: {btc_pct:.1%} (цель: 40%)")
        print(f"  ETH: {eth_pct:.1%} (цель: 20%)")
        
        assert abs(btc_pct - 0.40) < 0.05, "BTC распределение отклоняется от цели"
        assert abs(eth_pct - 0.20) < 0.05, "ETH распределение отклоняется от цели"
        
        print("✓ Распределение проверено!")

### 4.1 Фреймворк бэктестирования

```python
# research/backtest_engine.py

class BacktestEngine:
    def __init__(self, start_date, end_date, initial_capital):
        self.start_date = start_date
        self.end_date = end_date
        self.capital = initial_capital
        self.positions = {}
        self.trades = []
        
    def load_data(self):
        """
        Загрузка исторических данных из Bybit
        """
        self.btc_data = fetch_historical_ohlcv(
            symbol='BTCUSDT',
            start=self.start_date,
            end=self.end_date,
            timeframe='1h'
        )
        
        self.eth_data = fetch_historical_ohlcv(
            symbol='ETHUSDT',
            start=self.start_date,
            end=self.end_date,
            timeframe='1h'
        )
        
    def run_trend_strategy(self, params):
        """
        Бэктест стратегии следования тренду
        """
        results = {
            'trades': [],
            'equity_curve': [],
            'drawdowns': []
        }
        
        equity = self.capital
        max_equity = equity
        
        for i in range(200, len(self.btc_data)):
            # Расчёт индикаторов
            window = self.btc_data[i-200:i]
            sma_50 = calculate_sma(window['close'], 50)
            sma_200 = calculate_sma(window['close'], 200)
            adx = calculate_adx(window, 14)
            
            current_price = self.btc_data[i]['close']
            
            # Проверка входа
            if (current_price > sma_200[-1] and 
                sma_50[-1] > sma_200[-1] and 
                adx[-1] > 25 and
                'BTC' not in self.positions):
                
                # Вход в лонг
                atr = calculate_atr(window, 14)[-1]
                position_size = self.calculate_position_size(
                    equity, current_price, atr
                )
                
                self.positions['BTC'] = {
                    'entry_price': current_price,
                    'size': position_size,
                    'stop_loss': current_price - (2 * atr)
                }
                
            # Проверка выхода
            elif 'BTC' in self.positions:
                position = self.positions['BTC']
                
                if current_price < sma_200[-1]:
                    # Выход при пробое тренда
                    pnl = (current_price - position['entry_price']) * position['size']
                    equity += pnl
                    
                    results['trades'].append({
                        'entry': position['entry_price'],
                        'exit': current_price,
                        'pnl': pnl,
                        'pnl_pct': (current_price - position['entry_price']) / position['entry_price']
                    })
                    
                    del self.positions['BTC']
            
            # Отслеживание капитала
            results['equity_curve'].append({
                'timestamp': self.btc_data[i]['timestamp'],
                'equity': equity
            })
            
            # Отслеживание просадки
            if equity > max_equity:
                max_equity = equity
            drawdown = (max_equity - equity) / max_equity
            results['drawdowns'].append(drawdown)
        
        return results
    
    def calculate_metrics(self, results):
        """
        Расчёт метрик производительности
        """
        trades = results['trades']
        equity_curve = results['equity_curve']
        
        winning_trades = [t for t in trades if t['pnl'] > 0]
        losing_trades = [t for t in trades if t['pnl'] <= 0]
        
        total_return = (equity_curve[-1]['equity'] - self.capital) / self.capital
        max_drawdown = max(results['drawdowns'])
        
        metrics = {
            'total_return': total_return,
            'max_drawdown': max_drawdown,
            'win_rate': len(winning_trades) / len(trades) if trades else 0,
            'avg_win': np.mean([t['pnl'] for t in winning_trades]) if winning_trades else 0,
            'avg_loss': np.mean([t['pnl'] for t in losing_trades]) if losing_trades else 0,
            'profit_factor': (
                sum(t['pnl'] for t in winning_trades) / 
                abs(sum(t['pnl'] for t in losing_trades))
            ) if losing_trades else float('inf'),
            'sharpe_ratio': self.calculate_sharpe(equity_curve),
            'total_trades': len(trades)
        }
        
        return metrics
```
